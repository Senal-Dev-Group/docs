## 第二步：构建请求签名

每个 REST 请求需携带以下 Headers：

| Header | 说明 |
|--------|-------------|
| `SENAL-API-KEY` | 你的 API Key |
| `SENAL-SIGNATURE` | HMAC-SHA256 签名 |
| `SENAL-TIMESTAMP` | 当前时间戳（毫秒） |
| `SENAL-NONCE` | 随机字符串（防重放） |
| `SENAL-PASSPHRASE` | 你的 Passphrase |

**签名算法：**

```
message = timestamp + nonce + METHOD + path + SHA256(body)
signature = HMAC-SHA256(apiSecret, message)
```

<CodeGroup>
```typescript TypeScript
import { createHmac, createHash, randomBytes } from 'crypto';

function buildHeaders(
  apiKey: string,
  apiSecret: string,
  passphrase: string,
  method: string,
  path: string,
  body: string = '',
) {
  const timestamp = Date.now().toString();
  const nonce = randomBytes(16).toString('hex');
  const bodyHash = createHash('sha256').update(body).digest('hex');
  const message = `${timestamp}${nonce}${method.toUpperCase()}${path}${bodyHash}`;
  const signature = createHmac('sha256', apiSecret).update(message).digest('hex');

  return {
    'SENAL-API-KEY': apiKey,
    'SENAL-SIGNATURE': signature,
    'SENAL-TIMESTAMP': timestamp,
    'SENAL-NONCE': nonce,
    'SENAL-PASSPHRASE': passphrase,
    'Content-Type': 'application/json',
  };
}
```

```python Python
import hmac
import hashlib
import time
import os

def build_headers(api_key, api_secret, passphrase, method, path, body=""):
    timestamp = str(int(time.time() * 1000))
    nonce = os.urandom(16).hex()
    body_hash = hashlib.sha256(body.encode()).hexdigest()
    message = f"{timestamp}{nonce}{method.upper()}{path}{body_hash}"
    signature = hmac.new(
        api_secret.encode(), message.encode(), hashlib.sha256
    ).hexdigest()

    return {
        "SENAL-API-KEY": api_key,
        "SENAL-SIGNATURE": signature,
        "SENAL-TIMESTAMP": timestamp,
        "SENAL-NONCE": nonce,
        "SENAL-PASSPHRASE": passphrase,
        "Content-Type": "application/json",
    }
```

## 第三步：提交首笔订单

<CodeGroup>
```typescript TypeScript
import axios from 'axios';

const BASE_URL = 'https://api.senal.market/v1';
const API_KEY = 'your-api-key';
const API_SECRET = 'your-api-secret';
const PASSPHRASE = 'your-passphrase';

async function placeOrder() {
  const path = '/v1/mm/orders';
  const body = JSON.stringify({
    marketID: 'mkt_xxx',
    tokenID: 'out_xxx',
    side: 'buy',
    orderType: 'GTC',
    price: 0.55,
    size: 100,
  });

  const headers = buildHeaders(API_KEY, API_SECRET, PASSPHRASE, 'POST', path, body);
  const res = await axios.post(`${BASE_URL}/mm/orders`, JSON.parse(body), { headers });
  console.log('Order placed:', res.data);
}

placeOrder();
```

```python Python
import requests
import json

BASE_URL = "https://api.senal.market/v1"
API_KEY = "your-api-key"
API_SECRET = "your-api-secret"
PASSPHRASE = "your-passphrase"

def place_order():
    path = "/v1/mm/orders"
    body = json.dumps({
        "marketID": "mkt_xxx",
        "tokenID": "out_xxx",
        "side": "buy",
        "orderType": "GTC",
        "price": 0.55,
        "size": 100,
    })

    headers = build_headers(API_KEY, API_SECRET, PASSPHRASE, "POST", path, body)
    res = requests.post(f"{BASE_URL}/mm/orders", data=body, headers=headers)
    print("Order placed:", res.json())

place_order()
```
</CodeGroup>

## 第四步：启动心跳

做市期间需每 10 秒发送一次心跳，超过 15 秒未收到心跳将自动撤销所有挂单。

```typescript
setInterval(async () => {
  const path = '/v1/mm/heartbeats';
  const headers = buildHeaders(API_KEY, API_SECRET, PASSPHRASE, 'POST', path);
  await axios.post(`${BASE_URL}/mm/heartbeats`, {}, { headers });
}, 10_000);
```
